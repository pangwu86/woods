#---------------------------------------------------------
# owner 为指定用户
# "@zozoh"
$ow  : ^(@)([a-z]+[a-z0-9]{3,})$
	${2} = String
#---------------------------------------------------------
# 角色为指定用户
# "@A:zozoh"     # admin
$admin       : ^(@A:)([a-z]+[a-z0-9]{3,})$
	${2} = String
#---------------------------------------------------------
# "@M:zozoh"     # member
$member      : ^(@M:)([a-z]+[a-z0-9]{3,})$
	${2} = String
#---------------------------------------------------------
# "@C:zozoh"     # contributor
$contributor : ^(@C:)([a-z]+[a-z0-9]{3,})$
	${2} = String
#---------------------------------------------------------
# "@W:zozoh"     # watcher
$watcher     : ^(@W:)([a-z]+[a-z0-9]{3,})$
	${2} = String
#---------------------------------------------------------
# "@AM:zozoh"    # admin + member
$roles     : ^(@)([AMCW]{2,})(:)([a-z]+[a-z0-9]{3,})$
	{r:'${2}',nm:'${4}'} = Json
#---------------------------------------------------------
# 脱字符开头的表示正则表达式，匹配域名（不是别名）
# "^a.*"
$nm : ^([\^].+)$
	${1} = Regex
#---------------------------------------------------------
# 父域包含某指定域
# ">danoo"
$pa  : ^(>)([a-z0-9]+)$
    ${2} = String
#---------------------------------------------------------
# 祖先包含某指定域
# ">>danoo"
$ans :: >>
    ${2} = String
#---------------------------------------------------------
# 友域包含某指定域
# "->danoo"
$frs :: ->
    ${2} = String
#---------------------------------------------------------
# 按类型查询 
# "T(sin,org)"
$tp : ^(T[(])([A-Za-z, \t]*)([)])$
    ${2} = StringEnum
#---------------------------------------------------------
# 按域的模式 
# "M(PUBLIC,PRIVATE)"
$md : ^(M[(])([A-Za-z, \t]*)([)])$
    ${2} = StringEnum
#---------------------------------------------------------
# 按域的创建时间
# "C(2013-09-10 12:34:15, 2013-09-11 13:45:12)"  # 支持时间开闭区间
# "C(2013-09-10, 2013-08-23)"  # 会自动交换，根据开闭区间，给出时间
$ctm :: C
    ${2} = DateRegion
#---------------------------------------------------------
# 按域的最后修改时间
# "L[2013-09-11, 2013-07-25)"
$lm  :: L
    ${2} = DateRegion
#---------------------------------------------------------
# 按标签查询
# "#(酒店,餐饮)"
$lbs : ^(#[(])(.*)([)])$
    ${2} = StringEnum
#---------------------------------------------------------
# 活跃级别范围
# "lv(0,4)"      # dmn.live>0 && dmn.live<4
# "lv[0,4)"      # dmn.live>=0 && dmn.live<4
# "lv(0,)"       # dmn.live>0
# "lv(,4)"       # dmn.live<4
# "lv(0)"        # dmn.live==0
$lv  :: lv
    ${2} = IntRegion
#---------------------------------------------------------
# 普通字符串，表示域名或者别名包括这段文字的域
$text : ^(.*)$
    ${1} = String
